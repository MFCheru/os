void TaskBatch(int pid, vector<int> params) {

	srand(1); // set seed

	int total_cpu = params[0];
	int cant_bloqueos = params[1];

	for (int i = 0; i < total_cpu; ++i) {
		uso_CPU(pid, 1); // syscall or normal usage
		if (cant_bloqueos > 0 && rand() % 2 == 0) { // block
			uso_IO(pid, 1);
			cant_bloqueos--;
		}
	}
}
